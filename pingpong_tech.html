SQL · RLS 策略示例
-- 用户只能修改自己的档案
CREATE POLICY "users_update_own" ON users
  FOR UPDATE USING (auth.uid() = id);

-- 帖子：公开读，登录后可写
CREATE POLICY "posts_read_all" ON posts
  FOR SELECT USING (true);
CREATE POLICY "posts_insert_auth" ON posts
  FOR INSERT WITH CHECK (auth.uid() = author_id);

-- 等级自动计算触发器
CREATE OR REPLACE FUNCTION sync_level() RETURNS TRIGGER AS $$
BEGIN
  NEW.level := LEAST(100, GREATEST(1,
    FLOOR((NEW.kaiqiu_score - 1000) / 10)::int2));
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER on_kaiqiu_score_update
  BEFORE UPDATE OF kaiqiu_score ON users
  FOR EACH ROW EXECUTE FUNCTION sync_level();

// lib/api.ts — 统一响应格式
export type ApiResponse<T> = {
  data:    T | null
  error:   string | null
  status:  number
}

export function ok<T>(data: T): Response {
  return Response.json({ data, error: null, status: 200 })
}
export function err(msg: string, status = 400): Response {
  return Response.json({ data: null, error: msg, status }, { status })
}

// 开球网积分同步 Edge Function
export async function syncKaiqiuScore(userId: string) {
  const { data: user } = await supabase
    .from('users').select('kaiqiu_id').eq('id', userId).single()
  if (!user?.kaiqiu_id) return

  const score = await fetchKaiqiuScore(user.kaiqiu_id) // 调开球网 API
  await supabase.from('users').update({
    kaiqiu_score: score,
    kaiqiu_synced_at: new Date().toISOString()
  }).eq('id', userId) // 触发器自动计算 level
}
